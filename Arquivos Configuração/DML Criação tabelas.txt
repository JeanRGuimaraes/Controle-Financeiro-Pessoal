criação tabela divida

create table DIVIDAS(
CODIGO int not null,
DESCRICAO VARCHAR(200),
VALOR DOUBLE,
DTINICIAL VARCHAR(10),
DTFINAL VARCHAR(10),
STATUS VARCHAR(1),
COD_USUARIO INT,
PRIMARY KEY(CODIGO)
);

Tabela de PESSOAS

create table PESSOAS(
CODIGO int not null,
NOME VARCHAR(200),
RG NUMERIC,
GENERO VARCHAR(1),
EMAIL VARCHAR(50),
PROFISSAO VARCHAR(50),
TELEFONE VARCHAR(11),
CPF VARCHAR(11),
COD_USUARIO INT,
PRIMARY KEY(CODIGO)
);

CREATE TABLE RENDA (
CODIGO int not null,
NOME VARCHAR(50),
RENDA DOUBLE,
PERIODOINICIAL VARCHAR(10),
PERIODOFINAL VARCHAR(10),
OBSERVACOES VARCHAR(50),
COD_USUARIO INT,
PRIMARY KEY(CODIGO)
);

Create table USUARIO(
CODIGO INT,
NOME VARCHAR(20),
USUARIO VARCHAR(15),
SENHA VARCHAR(15),
PRIMARY KEY(CODIGO));
