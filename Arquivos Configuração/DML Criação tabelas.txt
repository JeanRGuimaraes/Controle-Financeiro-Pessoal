criação tabela divida

create table DIVIDAS(
CODIGO int not null,
DESCRICAO VARCHAR(200),
VALOR DOUBLE,
DTINICIAL VARCHAR(10),
DTFINAL VARCHAR(10),
STATUS VARCHAR(50),
PRIMARY KEY(CODIGO)
);

Tabela de PESSOA

create table PESSOA(
CODIGO int not null,
NOME VARCHAR(200),
RG NUMERIC,
GENERO VARCHAR(1),
EMAIL VARCHAR(50),
PROFISSAO VARCHAR(50),
TELEFONE VARCHAR(11),
CPF VARCHAR(11),
PRIMARY KEY(CODIGO)
);
